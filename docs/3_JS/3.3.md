```javascript
const stringA = prompt('a =');
const stringB = prompt('b =');

if(!isNumber(stringA)) {
    alert(`Неверное число a = ${stringA}`);
    return;
}
if(!isNumber(stringB)) {
    alert(`Неверное число b = ${stringB}`);
    return;
}

const a = parseFloat(stringA);
const b = parseFloat(stringB);

write(a, '+', b, '=', roundTo3(a + b));
write(a, '-', b, '=', roundTo3(a - b));
write(a, '*', b, '=', roundTo3(a * b));

if(isFinite(a / b)) {
    write(a, '/', b, '=', roundTo3(a / b));
} else {
    write(a, '/', b, '-', 'на ноль делить нельзя!');
}

function isNumber(str) {
    return !isNaN(parseFloat(str)) && isFinite(str);
}

function roundTo3(number) {
    return Math.round(number * 1000) / 1000;
}

function write(number1, operator, number2, resultOperator, result) {
    document.write(`<p>a ${operator} b = ${number1} ${operator} ${number2} ${resultOperator} ${result}</p>`);
}
```

## Задание 7
```javascript
const amountString = prompt('Введите сумму:');

if(!isNumber(amountString)) {
    alert(`Ошибочная сумма: ${amountString}`);
    return;
}

const money = getMoney(amountString);
const roubles = getRoubles(money);
const kopeyks = getKopeyks(money);

const roublesString = getRoublesString(roubles);
const kopeyksString = getKopeyksString(kopeyks);

document.write(`<p><b>Сумма:</b> ${roubles} ${roublesString} ${kopeyks} ${kopeyksString}.</p>`);

function isNumber(string) {
    return !isNaN(string) && isFinite(string);
}
function getMoney(string) {
    return Math.round(parseFloat(string) * 100) / 100;
}
function getRoubles(money) {
    return Math.floor(money);
}
function getKopeyks(money) {
    return Math.round((money - Math.floor(money)) * 100);
}
function getRoublesString(number) {
    const twoDigits = number % 100;
    const lastDigit = number % 10;
    
    if(11 <= twoDigits && twoDigits <= 14) {
        return 'рублей';
    }
    if(lastDigit === 1) {
        return 'рубль';
    }
    if(2 <= lastDigit && lastDigit <= 4) {
        return 'рубля';
    }
    
    return 'рублей';
}
function getKopeyksString(number) {
    const twoDigits = number % 100;
    const lastDigit = number % 10;
    
    if(11 <= twoDigits && twoDigits <= 14) {
        return 'копеек';
    }
    if(lastDigit === 1) {
        return 'копейка';
    }
    if(2 <= lastDigit && lastDigit <= 4) {
        return 'копейки';
    }
    
    return 'копеек';
}
```
