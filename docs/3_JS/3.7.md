```javascript
function unionUsers(users1, users2) {
    users1 = users1.map(u => u.toLowerCase());
    users2 = users2.map(u => u.toLowerCase());

    users2.forEach(eachUsers2 => { 
        if (users1.indexOf(eachUsers2) === -1) {
            let i = 0;
            while (i < users1.length && users1[i] < eachUsers2) { 
                i++;
            }
            users1.splice(i, 0, eachUsers2);
        } 
    });

    return users1;
}
```

```
function usersRating(arr) {
    const result = [];
    
    for(let i in arr) {
        const name = arr[i][0].toLowerCase();
        
        if(!result.find(e => e[0] === name)) {
            const sum = arr.filter(e => e[0].toLowerCase() === name).reduce((sum, e) => sum + e[1], 0);
            result.push([name, sum]);
        }
    }
    
    result.sort((e1, e2) => {
        if(e1[1] !== e2[1]) {
            return e2[1] - e1[1];
        }
    
        if(e1[0] !== e2[0]) {
            return (e1[0] > e2[0]) ? 1 : -1;
        }
    
        return 0;
    });
    
    return result;
}
```
