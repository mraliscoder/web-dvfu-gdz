function insertSpace(text, n) {
    let s = " ";
    s = s.repeat(n)
    let newStr = "";
    for (let i = 0; i < text.length - 1; i++) {
        let x = ""
        x = text[i] + s;
        newStr = newStr + x
    }
    newStr = newStr + text[text.length - 1] + "!";
    return newStr;
}
```

function getCountP(html) {
  html = html.toLowerCase();

  let start = 0;
  let count = 0;

  while (html.indexOf("<p ", start) >= 0) {
    start = html.indexOf("<p ", start) + 3;
    count++;
  }
  return count;
}
```
