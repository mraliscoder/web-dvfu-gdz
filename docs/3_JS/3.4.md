## Задание 3
```javascript
let n = prompt("n =");
let html = "";
if (
  isNaN(parseFloat(n)) ||
  !isFinite(n) ||
  n <= 0 ||
  parseInt(n) !== parseFloat(n)
) {
  alert(`Ошибочное число: ${n}`);

} else {
  n = +n;
  for (let i = 0; i < n; i++) {
    html += "<tr>";
    for (let j = 0; j < n; j++) {
      if (
        i == j ||
        i + j == n - 1 ||
        (i > j && i < n - 1 - j) ||
        (i < j && i > n - 1 - j)
      ) {
        html += '<td class="black"></td>';
      } else if ((i < j && i < n - 1 - j) || (i > j && i > n - 1 - j)) {
        html += '<td class="white"></td>';
      }
    }
    html += "</tr>";
  }
  document.write(`<table>${html}</table>`);
}
```
